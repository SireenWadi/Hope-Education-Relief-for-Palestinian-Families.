<script>
  // عرض التاريخ في الـ Top Bar
  const dateEl = document.getElementById('current-date');
  if (dateEl) {
    const now = new Date();
    const options = { day: '2-digit', month: 'short', year: 'numeric' };
    dateEl.textContent = now.toLocaleDateString('en-GB', options);
  }

  // تفعيل أزرار المبالغ السريعة في قسم Quick Donate
  const amountPills = document.querySelectorAll('.amount-pill');
  const customAmountInput = document.getElementById('customAmount');

  amountPills.forEach(pill => {
    pill.addEventListener('click', () => {
      amountPills.forEach(p => p.classList.remove('active'));
      pill.classList.add('active');
      const amount = pill.getAttribute('data-amount');
      if (customAmountInput) {
        customAmountInput.value = amount;
      }
    });
  });
   const phrases = [
    "Join us in empowering Palestinian students through scholarships and hope.",
    "Supporting orphans and stranded students with dignity and care.",
    "Providing food, shelter, and emergency relief for families in crisis."
  ];

  const typedElement = document.getElementById("typed-text");
  let phraseIndex = 0;
  let charIndex = 0;
  let isDeleting = false;

  function typeLoop() {
    const currentPhrase = phrases[phraseIndex];

    if (!isDeleting) {
      // كتابة
      typedElement.textContent = currentPhrase.substring(0, charIndex + 1);
      charIndex++;

      if (charIndex === currentPhrase.length) {
        // بعد ما يخلص الجملة، استنى شوي
        setTimeout(() => { isDeleting = true; }, 1400);
      }
    } else {
      // مسح
      typedElement.textContent = currentPhrase.substring(0, charIndex - 1);
      charIndex--;

      if (charIndex === 0) {
        isDeleting = false;
        phraseIndex = (phraseIndex + 1) % phrases.length;
      }
    }
const speed = isDeleting ? 50 : 90;

    setTimeout(typeLoop, speed);
  }

  document.addEventListener("DOMContentLoaded", () => {
    typeLoop();
  });
